lp_situation_growth = {
	picture = GFX_evt_friendly_enlightenment
	complete_icon_frame = GFX_situation_outcome_frame_green
	complete_icon = GFX_situation_outcome_positive
	category = positive

	on_start = {
		lp_set_growth_resource = yes
	}
	on_progress_complete = {
		custom_tooltip = lp_situation_growth_gain
		hidden_effect = {
			situation_event = { id = lp_situation.101 }
			owner = {
				capital_scope = {
					planet_event = { id = lp_planet.010 }
				}
			}
		}
	}
	
	stages = {
		lp_stage_growing = {
			icon = GFX_situation_stage_eat
			icon_background = GFX_situation_stage_frame_green
			end = {
				base = 6
				modifier = {
					add = owner.capital_scope.lp_growth_factor
				}
			}
		}
	}
	monthly_progress = {
		base = 1
		modifier = {
			mult = 0
			desc = "lp_situation_growth_pause_desc"
			owner = {
				has_lp_growth_resources = no
			}
		}
	}

	approach = {
		name = lp_situation_growth_approach_standard
		icon = GFX_situation_approach_research
		icon_background = GFX_situation_approach_bg_green
		default = yes

		on_select = {
			custom_tooltip = lp_situation_growth_approach_standard
		}

		modifier = {

		}
		resources = {
			category = country_situations
			upkeep = {
				lp_planetary_matter = 100
				mult = owner.value:lp_situation_growth_modifier
			}
		}
	}

	on_monthly = {
		events = {
			lp_situation.1
		}
	}

	triggered_blocked_desc = {
		trigger = {
			owner = {
				has_lp_growth_resources = no
			}
		}
		text = "lp_situation_growth_pause_desc"
	}
}
