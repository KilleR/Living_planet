lp_situation_lunar_shift = {
	picture = GFX_evt_circuitry_modification
	complete_icon_frame = GFX_situation_outcome_frame_green
	complete_icon = GFX_situation_outcome_positive
	category = positive

	on_start = {
		set_variable = {
			which = lp_lunar_shift_type
			value = generic
		}
		situation_event = { id = lp_situation.300 }
	}

	on_progress_complete = {
		owner = {
			set_country_flag = lp_lunar_shift_completed
		}
		switch = {
			trigger = has_situation_flag

			lp_lunar_flag_general = {
				owner.capital_scope = {
					add_deposit = d_living_planet_neural_net_general
				}
			}
			lp_lunar_flag_physics = {
				owner.capital_scope = {
					add_deposit = d_living_planet_neural_net_physics
				}
			}
			lp_lunar_flag_society = {
				owner.capital_scope = {
					add_deposit = d_living_planet_neural_net_society
				}
			}
			lp_lunar_flag_engineering = {
				owner.capital_scope = {
					add_deposit = d_living_planet_neural_net_engineering
				}
			}
		}
		destroy_situation = this
	}

	stages = {
		lp_stage_evo_init = {
			icon = GFX_situation_stage_1
			icon_background = GFX_situation_stage_frame_green
			end = 25
		}
		lp_stage_lunar_change = {
			icon = GFX_situation_stage_2
			icon_background = GFX_situation_stage_frame_green
			end = 50
			on_first_enter = {
				situation_event = {
					id = lp_situation.301
				}
			}
		}
	}

	monthly_progress = {
		base = 1
	}
	approach = {
		name = lp_evo_approach_research
		icon = GFX_situation_approach_research
		icon_background = GFX_situation_approach_bg_green
		default = yes
		on_select = {
			custom_tooltip_with_params = {
				description = lp_situation_evo_shift_approach
				description_parameters = {
					APPROACH = "research"
				}
			}
		}
		modifier = {
			country_physics_research_produces_mult = -0.5
			country_society_research_produces_mult = -0.5
			country_engineering_research_produces_mult = -0.5
		}
	}
	approach = {
		name = lp_evo_approach_physics
		icon = GFX_situation_approach_physics
		icon_background = GFX_situation_approach_bg_green
		on_select = {
			custom_tooltip_with_params = {
				description = lp_situation_evo_shift_approach
				description_parameters = {
					APPROACH = "physics"
				}
			}
		}
		modifier = {
			country_physics_research_produces_mult = -1
		}
	}
	approach = {
		name = lp_evo_approach_society
		icon = GFX_situation_approach_society
		icon_background = GFX_situation_approach_bg_green
		on_select = {
			custom_tooltip_with_params = {
				description = lp_situation_evo_shift_approach
				description_parameters = {
					APPROACH = "society"
				}
			}
		}
		modifier = {
			country_society_research_produces_mult = -1
		}
	}
	approach = {
		name = lp_evo_approach_engineering
		icon = GFX_situation_approach_engineering
		icon_background = GFX_situation_approach_bg_green
		on_select = {
			custom_tooltip_with_params = {
				description = lp_situation_evo_shift_approach
				description_parameters = {
					APPROACH = "engineering"
				}
			}
		}
		modifier = {
			country_engineering_research_produces_mult = -1
		}
	}
}
