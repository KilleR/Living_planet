decision_lp_devour_biomass = {
	icon = decision_planet_beam
	owned_planets_only = no
	
	potential = {
		is_colonizable = yes
		NOT = { has_modifier = living_planet_devoured }
		exists = space_owner
		space_owner = {
			is_same_value = from
			has_origin = origin_living_planet
		}
	}
	
	effect = {
		add_modifier = { modifier = living_planet_devoured }
		
		solar_system = {
			spawn_megastructure = {
				type = planet_devourer
				planet = ROOT
				owner = space_owner
			}
		}
	}
}
