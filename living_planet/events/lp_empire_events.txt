namespace = lp_empire

# Calculate devouring growth rate
country_event = {
	id = lp_empire.001
	hide_window = yes

	is_triggered_only = yes

	immediate = {
		if = {
			limit = {
				count_situation = {
					limit = {
						is_situation_type = lp_situation_growth
					}
					count = 0
				}
			}
			lp_start_growth_situation = yes
		}

		every_owned_megastructure = {
			limit = {
				is_megastructure_type = planet_devourer
			}
			save_event_target_as = event_devourer_megastructure

			if = {
				limit = {
					NOT = {
						exists = planet
					}
				}
				remove_megastructure = event_target:event_devourer_megastructure
			}
		}
	}
}
