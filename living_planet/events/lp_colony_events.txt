namespace = lp_colony

# Colony Creation
country_event = {
	id = lp_colony.001
	hide_window = yes

	is_triggered_only = yes

	trigger = {
		has_origin = origin_living_planet
		is_ai = no
	}
	immediate = {
		capital_scope = {
			clear_blockers = yes
			while = {
				count = 3
				add_district = {
					district_type = district_hive_1
				}
			}
			add_zone = {
				district = district_hive_1
				zone = zone_energy_hive
			}
			while = {
				count = 2
				add_district = {
					district_type = district_hive_2
				}
			}
			add_zone = {
				district = district_hive_2
				zone = zone_minerals_hive
			}
			while = {
				count = 4
				add_district = {
					district_type = district_hive_3
				}
			}
			add_zone = {
				district = district_hive_3
				zone = zone_food_hive
			}

			add_blocker = {
				type = d_living_planet_proto_hive
			}
		}
	}
}
