# Planet devouring
event = {
	id = lp_planet.001
	hide_window = yes
	
	is_triggered_only = yes
	
	trigger = {
		any_country = {
			has_origin = origin_living_planet
			any_megastructure = {
				is_megastructure_type = planet_devourer
			}
		}
	}
	
	immediate = {
		every_country = {
			limit = {
				any_owned_megastructure = {
					is_megastructure_type = planet_devourer
				}
			}
			every_owned_megastructure = {
				limit = {
					is_megastructure_type = planet_devourer
				}
				save_event_target_as = event_devourer_megastructure
				
				if = {
					limit = {
						NOT = {
							exists = planet
						}
					}
					remove_megastructure = event_target:event_devourer_megastructure
				}
				else = {
					planet = {
						save_event_target_as = event_devoured_planet
						space_owner = {
							capital_scope = {
								if = {
									limit = {
										event_target:event_devoured_planet = {
											is_artificial = yes
										}
									}
									add_deposit = d_living_planet_megastructure
									change_variable = {
										which = biomass_acquired
										value = 0.5
									}
								}
								else = {
									change_variable = {
										which = biomass_acquired
										value = 1
									}
								}
								save_event_target_as = event_capital
								if = {
									limit = {
										check_variable = {
											which = biomass_acquired
											value >= lp_growth_factor
										}
									}
									# grow the planet
									planet_event = { id = lp_planet.010 }
									subtract_variable = {
										which = biomass_acquired
										value = lp_growth_factor
									}
								}
							}
						}
						
						if = {
							limit = {
								planet_size > 1
							}
							change_planet_size = -1
						} else = {
							remove_planet = yes
							remove_megastructure = event_target:event_devourer_megastructure
						}
						
						random_deposit = {
							save_event_target_as = deposit
							if = {
								limit = {
									has_deposit_category = deposit_cat_minerals
								}
								event_target:event_capital = {
									add_deposit = d_living_planet_minerals
								}
							}
							if = {
								limit = {
									has_deposit_category = deposit_cat_energy
								}
								event_target:event_capital = {
									add_deposit = d_living_planet_energy
								}
							}
							if = {
								limit = {
									has_deposit_category = deposit_cat_food
								}
								event_target:event_capital = {
									add_deposit = d_living_planet_food
								}
							}
							if = {
								limit = {
									OR = {
										has_deposit_category = deposit_cat_rare
										has_deposit_category = deposit_cat_rare_blocker
									}
								}
								event_target:event_capital = {
									add_deposit = d_living_planet_rare
								}
							}
							if = {
								limit = {
									has_deposit_category = deposit_cat_blockers
								}
								event_target:event_capital = {
									add_deposit = d_living_planet_unity
								}
							}
							remove_deposit = yes
						}
					}
				}
			}
		}
		
	}
}


# Planet changing ownership
planet_event = {
	id = lp_planet.002
	hide_window = yes
	
	is_triggered_only = yes
	
	trigger = {
		this = {
			is_capital = no
		}
		from = {
			has_origin = origin_living_planet
		}
		this = {
			any_pop = {
				has_trait = trait_living_planet
			}
		}
	}
	
	immediate = {
		this = {
			every_owned_pop = {
				limit = {
					has_trait = trait_living_planet
				}
				kill_pop = yes
			}
		}
	}
}

# Displace a planet pop to the Living Planet
event = {
	id = lp_planet.004
	
	is_triggered_only = yes
	hide_window = yes
	
	trigger = {
		any_country = {
			has_origin = origin_living_planet
			any_owned_planet = {
				is_capital = no
				num_pops > 0
			}
		}
	}
	
	immediate = {
		every_country = {
			limit = {
				has_origin = origin_living_planet
				any_owned_planet = {
					is_capital = no
					num_pops > 0
				}
			}
			
			every_owned_planet = {
				limit = {
					is_capital = no
					num_pops > 0
				}
				
				random_owned_pop = {
					create_message = {
						type = MESSAGE_TYPE_LP_POP_RELOCATED
						localization = MESSAGE_LP_POP_RELOCATED
						days = 30
						target = this.planet
						variable = {
							type = name
							localization = SPECIES
							scope = this.species
						}
						variable = {
							type = name
							localization = ORIGIN
							scope = this.planet
						}
						variable = {
							type = name
							localization = CAPITAL
							scope = this.owner.capital_scope
						}
					}
					resettle_pop = {
						pop = this
						planet = this.owner.capital_scope
					}
				}
			}
		}
	}
	
}

# Evo shift
planet_event = {
	id = lp_planet.011
	title = lp_planet.011.name
	desc = lp_planet.011.desc
	is_triggered_only = yes
	picture = GFX_evt_genetic_modification
	
	immediate = {
		change_variable = {
			which = lp_evolutionary_shift
			value = 25
		}
		change_variable = {
			which = lp_growth_factor
			value = 1
		}
	}
	
	option = {
		name = lp_planet.011.a
		custom_tooltip = lp_planet.011.a.tooltip
		if = {
			limit = {
				check_variable = {
					which = lp_evolutionary_shift
					value >= 75
				}
				NOT = {
					has_planet_flag = lp_lunar_shift_done
				}
			}
			set_planet_flag = lp_lunar_shift_done
			begin_event_chain = {
				event_chain = lp_lunar_shift_chain
				target = ROOT
			}
			enable_special_project = {
				name = LP_LUNAR_SHIFT_GENERAL
				location = this
				owner = root
			}
			enable_special_project = {
				name = LP_LUNAR_SHIFT_PHYSICS
				location = this
				owner = root
			}
			enable_special_project = {
				name = LP_LUNAR_SHIFT_SOCIETY
				location = this
				owner = root
			}
			enable_special_project = {
				name = LP_LUNAR_SHIFT_ENGINEERING
				location = this
				owner = root
			}
		}
		else_if = {
			limit = {
				check_variable = {
					which = lp_evolutionary_shift
					value >= 150
				}
				NOT = {
					has_planet_flag = lp_system_shift_done
				}
			}
			set_planet_flag = lp_system_shift_done
			begin_event_chain = {
				event_chain = lp_system_shift_chain
				target = ROOT
			}
		}
		else = {
			begin_event_chain = {
				event_chain = lp_evo_shift_chain
				target = ROOT
			}
			if = {
				limit = {
					check_variable = {
						which = lp_growth_factor
						value > 5
					}
				}
				enable_special_project = {
					name = LP_EVO_SHIFT_GENERAL
					location = this
					owner = root
				}
			}
			enable_special_project = {
				name = LP_EVO_SHIFT_PHYSICS
				location = this
				owner = root
			}
			enable_special_project = {
				name = LP_EVO_SHIFT_SOCIETY
				location = this
				owner = root
			}
			enable_special_project = {
				name = LP_EVO_SHIFT_ENGINEERING
				location = this
				owner = root
			}
			enable_special_project = {
				name = LP_EVO_SHIFT_ENERGY
				location = this
				owner = root
			}
			enable_special_project = {
				name = LP_EVO_SHIFT_MINERALS
				location = this
				owner = root
			}
			enable_special_project = {
				name = LP_EVO_SHIFT_FOOD
				location = this
				owner = root
			}
		}
	}
}
