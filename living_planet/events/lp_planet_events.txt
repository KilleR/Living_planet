namespace = lp_planet

# Planet devouring
event = {
	id = lp_planet.001
	hide_window = yes
	
	is_triggered_only = yes
	
	trigger = {
		any_country = {
			has_origin = origin_living_planet
		}
	}
	
	immediate = {
		every_country = {
			limit = {
				any_owned_megastructure = {
					is_megastructure_type = planet_devourer
				}
			}
			every_owned_megastructure = {
				limit = {
					is_megastructure_type = planet_devourer
				}
				save_event_target_as = event_devourer_megastructure
				
				if = {
					limit = {
						NOT = {
							exists = planet
						}
					}
					remove_megastructure = event_target:event_devourer_megastructure
				}
				else = {
					planet = {
						space_owner = {
							capital_scope = {
								save_event_target_as = event_capital
							}
						}

						random_deposit = {
							save_event_target_as = deposit
							if = {
								limit = {
									has_deposit_category = deposit_cat_minerals
								}
								event_target:event_capital = {
									add_deposit = d_living_planet_minerals
								}
							}
							if = {
								limit = {
									has_deposit_category = deposit_cat_energy
								}
								event_target:event_capital = {
									add_deposit = d_living_planet_energy
								}
							}
							if = {
								limit = {
									has_deposit_category = deposit_cat_food
								}
								event_target:event_capital = {
									add_deposit = d_living_planet_food
								}
							}
							if = {
								limit = {
									OR = {
										has_deposit_category = deposit_cat_rare
										has_deposit_category = deposit_cat_rare_blocker
										has_deposit_category = deposit_cat_rare_resource
									}
								}
								event_target:event_capital = {
									add_deposit = d_living_planet_rare
								}
							}
							if = {
								limit = {
									OR = {
										has_deposit_category = deposit_cat_blockers
										has_deposit_category = deposit_cat_blockers_natural
										has_deposit_category = deposit_cat_misc
									}
								}
								event_target:event_capital = {
									add_deposit = d_living_planet_unity
								}
							}
							remove_deposit = yes
						}
						
						if = {
							limit = {
								planet_size > 1
							}
							change_planet_size = -1
						} else = {
							remove_planet = yes
							remove_megastructure = event_target:event_devourer_megastructure
						}
					}
				}
			}
		}
		
		every_country = {
			limit = {
				has_origin = origin_living_planet
			}
			country_event = { id = lp_empire.001 }
		}
		
	}
}


# Planet changing ownership
planet_event = {
	id = lp_planet.002
	hide_window = yes
	
	is_triggered_only = yes
	
	trigger = {
		this = {
			is_capital = no
		}
		from = {
			has_origin = origin_living_planet
		}
		this = {
			any_owned_pop_group = {
				has_trait = trait_living_planet
			}
		}
	}
	
	immediate = {
		this = {
			every_owned_pop_group = {
				limit = {
					has_trait = trait_living_planet
				}
				kill_pop_group = {
					pop_group = this
				}
			}
		}
	}
}

# Displace a planet pop to the Living Planet
event = {
	id = lp_planet.004
	
	is_triggered_only = yes
	hide_window = yes
	
	trigger = {
		any_country = {
			has_origin = origin_living_planet
			any_owned_planet = {
				is_capital = no
				any_owned_pop_group = {
					pop_group_size > 0
				}
			}
		}
	}
	
	immediate = {
		every_country = {
			limit = {
				has_origin = origin_living_planet
				any_owned_planet = {
					is_capital = no
					any_owned_pop_group = {
						pop_group_size > 0
					}
				}
			}
			
			every_owned_planet = {
				limit = {
					is_capital = no
					any_owned_pop_group = {
						pop_group_size > 0
					}
				}
				
				random_owned_pop_group = {
					create_message = {
						type = MESSAGE_TYPE_LP_POP_RELOCATED
						localization = MESSAGE_LP_POP_RELOCATED
						days = 30
						target = this.planet
						variable = {
							type = name
							localization = SPECIES
							scope = this.species
						}
						variable = {
							type = name
							localization = ORIGIN
							scope = this.planet
						}
						variable = {
							type = name
							localization = CAPITAL
							scope = this.owner.capital_scope
						}
					}

					resettle_pop_group = {
						POP_GROUP = this
						AMOUNT = 100
						PLANET = this.owner.capital_scope
					}
				}
			}
		}
	}
	
}

# Living Planet growth + Evolutionary Shifts
planet_event = {
	id = lp_planet.010
	hide_window = yes
	is_triggered_only = yes
	trigger = {
		always = yes
	}
	
	immediate = {
		change_planet_size = 1

		if = {
			limit = {
				check_variable = {
					which = lp_evolutionary_shift
					value = trigger:planet_size
				}
			}

			planet_event = { id = lp_planet.011 }

			owner.capital_scope = {
				change_variable = {
					which = lp_evolutionary_shift
					value = 10
				}
				change_variable = {
					which = lp_growth_factor
					value = 1
				}
			}
		}
	}
}
