# Fired from on_action 'on_monthly_pulse'
event = {
    id = dream.101
    hide_window = yes
    is_triggered_only = yes

    trigger = {
        any_playable_country = {
            has_ascension_perk = ap_dreaming_deep
        }
    }

    immediate = {
        every_playable_country = {
            if = {
                limit = {
                    NOT = { is_variable_set = dreamer_motes_produced }

                }
                set_variable = {
                    which = dreamer_motes_produced
                    value = 0
                }
            }

            export_resource_income_to_variable = {
                resource = dream_motes
                variable = dreamer_motes_produced
            }

            set_variable = {
                which = alien_dreamer_motes_produced
                value = 0
            }
        }
    }
}

event = {
    id = dream.102
    hide_window = yes
    is_triggered_only = yes

    trigger = {
        any_playable_country = {
            has_ascension_perk = ap_dreaming_deep
        }
    }

    immediate = {
        every_playable_country = {
            limit = {
                NOT = { has_ascension_perk = ap_dreaming_deep }
            }
            # Iterate every dreamer empire, and add this empire's mote production to their alien mote track
            every_playable_country = {
                limit = {
                    has_ascension_perk = ap_dreaming_deep
                }
                change_variable = {
                    which = alien_dreamer_motes_produced
                    value = prev.dreamer_motes_produced
                }
            }
        }
        every_playable_country = {
            limit = {
                has_ascension_perk = ap_dreaming_deep
            }
            divide_variable = {
                which = alien_dreamer_motes_produced
                value = 2
            }
            complete_crisis_objective = crisobj_dreamers
            complete_crisis_objective = crisobj_dreamers_alien
        }
    }
}

event = {
    id = dream.103
    hide_window = yes
    is_triggered_only = yes

    trigger = {
        any_playable_country = {
            any_situation = {
                is_situation_type = situation_dreaming_deep
            }
        }
    }

    immediate = {
        every_playable_country = {
            IF = {
                limit = {
                    NOT = { is_variable_set = dreamerStrength }
                }

                set_variable = {
                    which = dreamerStrength
                    value = 1
                }
            }

            change_variable = {
                which = dreamerStrength
                value = 0.1
            }


            every_owned_planet = {
                limit = {
                    has_modifier = dreaming_deep
                }
                add_modifier = {
                    modifier = dreaming_deep
                    mult = owner.dreamerStrength
                }
            }
        }
    }
}
