crisis_dreaming_level_1 = {
	# REQUIREMENTS
	allow = {
		custom_tooltip = {
		}
	}
	required_crisis_currency = 0

	# REWARDS
	perks = {
		menp_dream_sleepers_1
		menp_dream_espionage_1
		menp_dream_objectives_1
	}

	on_unlock = {
		add_dreaming_level_1_perks = yes
		hidden_effect = {
			owner = {
				begin_event_chain = {
					event_chain = dreaming_chain
					target = root
				}
				add_event_chain_counter = {
					event_chain = "dreaming_chain"
					counter = "dreaming_level_reached"
					amount = 1
				}
				country_event = { id = dream.001 }
				enable_special_project = {
					name = "DREAM_CRISIS_SPECIAL_PROJECT_1"
					owner = ROOT
				}
			}
		}
	}
}
